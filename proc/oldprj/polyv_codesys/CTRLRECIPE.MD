	ID:	INT; 				// ідентифікатор

	CMD: INT;       		// команда (ідентичний до команди процедури)

	BATCHID2: UDINT;		// формується на початку виконання рецепту з обладнання яке використовується в рецепті - 16#0R0T_0UMM, R – номер рецептурного елементу, T – номер теплиці, U – номер апарату з добривами (0 – без, 1 – DS3, 2 - DS4), MM – номер майстер рецепту (0 – створений на базі пустого рецепту) 
	STEP1: UINT;			// крок
	TSTEP1: UDINT;			// час кроку
	STEP2: UINT;			// крок 2
	TSTEP2: UDINT;			// час кроку 2
	PHU_MIX: PHASE_CTRL;    // перелік етапів процедур обладнання якими може керувати рецепт
	PHU_GRN: PHASE_CTRL;    // перелік етапів процедур обладнання якими може керувати рецепт
	PH_ROUTE: PHASE_CTRL;    // перелік етапів процедур обладнання якими може керувати рецепт
	PH_WTRDOWN: PHASE_CTRL;	    // перелік етапів процедур обладнання якими може керувати рецепт
	PH_WTRACID: PHASE_CTRL;
	PHU_MIX_HMI: PHASE_HMI;    // перелік етапів процедур обладнання якими може керувати рецепт
	PHU_GRN_HMI: PHASE_HMI;    // перелік етапів процедур обладнання якими може керувати рецепт
	PH_ROUTE_HMI: PHASE_HMI;    // перелік етапів процедур обладнання якими може керувати рецепт
	PH_WTRDOWN_HMI: PHASE_HMI;    // перелік етапів процедур обладнання якими може керувати рецепт	
	PH_WTRACID_HMI: PHASE_HMI;    // перелік етапів процедур обладнання якими може керувати рецепт
	EQUIMENTS_ALLOC_REQ: INT;//біти запитів на займання обладнання
	EQUIMENTS_ALLOC_CMPLT: INT;//біти результатів на займання обладнання
END_STRUCT
